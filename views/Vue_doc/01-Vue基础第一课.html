# Vue基础第一课

Vue是一套渐近式框架，是用于构建用户界面的一套框架，我们可以通过Vue来开发我们的单页面应用

# 创建Vue对象

var app = new Vue({
    el: "#box", //element   代表当前的vue对象接管的页面区域
    data: {
        userName: "杨标"
    }
})

# 元素的计算属性

我们经过刚刚上面的学习以后，我们都知道Vue里面两个非常重要的参数，一个是el:,一个是data

页面上面所有的数据都要经过data来接管，但是并不是所有的数据都要经过data来接管，有一种数据是可以不经过data来接管的

<body>
    <div id="app">
        <ul>
            <li v-for="(item,index) in procList">
                商品名称：<b>{{item.name}}</b> 商品价格：<b>{{item.price}}</b>
            </li>
        </ul>
        <hr>
        <div>
            <!-- 在这里，显示商品的总的价格-->
            总价：￥{{total}}
        </div>
    </div>
</body>
<script>
    var app = new Vue({
        el: "#app",
        data: {
            procList: [{
                name: "苹果手机",
                price: 4999
            }, {
                name: "小米手机",
                price: 2999
            }, {
                name: "大白菜",
                price: 4.5
            }]
        },
        computed: {
            //里面的每一个计算属性，其实就是一个方法，这个方法一定要有返回值
            total: function() {
                //在这里，我们需要把上面的proList里面的所有商品拿出来遍历一次，计算总价
                //第一件事情，拿到上面data里面的数据
                //我需要调用this.$data.属性   ----传统方式
                //新的方式 this.属性名
                //第二件事件，将值计算以后拿出去  return 
                var sum = 0;
                this.procList.forEach(item => {
                    sum += item.price;
                })
                return sum;
            }
        }
    })
</script>

# 观察属性【监听属性】

在Vue里面，我们可以动态的去监听一个值的变化，我们使用watch这个属性去监听，里面监听的每一项都是我们data里面的属性，也可以是我们computed里面的属性，然后后面跟一个方法，具体看下面代码：

<body>
    <div id="app">
        <input type="text" v-model="age">
        <span v-show="flag">未满18岁的人是不允许进网吧上网的</span>
        <!-- 
            未满18岁的人是不允许进网吧上网的
        -->
    </div>
</body>
<script>
    var app = new Vue({
        el: "#app",
        data: {
            age: 18,
            flag: false
        },
        watch: {
            age: function(newValue) {
                //这个newValue就是age的新的值
                if (newValue < 18) {
                    this.flag = true;
                } else {
                    this.flag = false;
                }
            }
        }
    })
</script>

# 对象属性的监听【深度监听】

在Vue里面，我们的普通监听只能够去监听我们的基本类型数据的属性，如果要监听一个对象，这个时候我们就需要使用深度监听，代码如下：

<body>
    <div id="app">
        <input type="text" v-model="userInfo.age">
        <span v-show="flag">未满18岁的人是不允许进网吧上网的</span>
        <hr>
        <div>{{userInfo.age}}</div>
        <!-- 
            未满18岁的人是不允许进网吧上网的
        -->
    </div>
</body>
<script>
    var app = new Vue({
        el: "#app",
        data: {
            userInfo: {
                name: "杨标",
                age: 18
            },
            flag: false
        },
        watch: {
            userInfo: {
                handler: function(newValue) {
                    console.log(newValue);
                },
                deep: true //深度监听
            }
        }
    })
</script>

# Vue当中的过滤器

在Vue的属性绑定过程当中，我们有时候需要对一些绑定的值做二次处理，虽然我们可以使用计算属性(computed)来处理这个问题，但是，我们不可能让所有的类型相同的属性都通过计算属性来完成，这样太浪费时间了，这个时候，我们可以使用Vue里面的过滤器（格式化）来完成这个操作，具体请看下面代码 ：

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>vue第二课</title>
    <script type="text/javascript" src="./js/vue.min.js"></script>
</head>

<body>
    <div class="app">
        <p>要求：2018-03-10 15:43:06</p>
        <hr>
        <div>开始日期：{{dateNow|toDate}}---------结束日期：{{endDate|toDate}}</div>
        <hr>
    </div>
</body>
<script>
    //如果要判断某个值是否是一个对象new出来的实例
    //var xxx=new Xxx();   var d=new Date()  var p=new Person();

    Number.prototype.toPadLeft = function(str) {
        //this 就相当于上面的num
        //str代表要补位的格式
        if (this.toString().length >= str.length) {
            return this.toString();
        } else {
            var count = str.length - this.toString().length; //要补位的长度
            var prefixerStr = ""; //存放要补多少个0
            for (var i = 0; i < count; i++) {
                prefixerStr += "0";
            }
            return prefixerStr + this.toString();
        }
    }
    //这个参数就是要格式化的值    下面的filter就是我们定义的过滤器
    //这一种定义方式 ，我们称之为全局过滤器
    Vue.filter("toDate", function(input) {
        //判断一下这个地方的input的类型 ，如果是date对象，我什么都不做 如果是字符串 我就要转成date对象
        var d = null;
        if (input instanceof Date) {
            //说明 它是通过Date 来new 出来的实例
            d = input;
        } else {
            //就有可能是string类型
            d = new Date(input);
        }
        var year = d.getFullYear();
        var month = d.getMonth().toPadLeft("00");
        var date = d.getDate().toPadLeft("00");

        var hour = d.getHours().toPadLeft("00");
        var minute = d.getMinutes().toPadLeft("00");
        var second = d.getSeconds().toPadLeft("00");

        //要求：2018-03-10 15:43:06
        //return `${year}-${month}-${date} ${hour}:${minute}:${second}`;
        return [year, month, date].join("-") + " " + [hour, minute, second].join(":");
    })
    var app = new Vue({
        //在Vue1.0的版本里面，它可以是html,也可以是body标签，但是在2.0里面，取消了
        //EL它是一个选择器，但是一定要注意，不能是body
        el: ".app",
        data: {
            dateNow: new Date(),
            endDate: "Tue Apr 10 2018 15:53:11 GMT+0800 (中国标准时间)"
        },
        computed: {

        }
    })
</script>
</html>

说明：在上面的代码当中，我们使用了Vue.filter这个方法云格式化我们的日期，让我们的日期在页面上面显示自己所需要的格式，这一种Vue.filter定义的过滤器，我们叫全局过滤器！

全局过滤器定义好了以后，直接使用就可以了，不需要其它配置，与全局过滤器相对应还有一种局部过滤器，局部过滤器如下所示：

<body>
    <div class="app">
        <p>要求：2018-03-10 15:43:06</p>
        <hr>
        <div>开始日期：{{dateNow|toDate}}---------结束日期：{{endDate|toDate}}</div>
        <hr>
    </div>
</body>
<script>
    //如果要判断某个值是否是一个对象new出来的实例
    //var xxx=new Xxx();   var d=new Date()  var p=new Person();

    Number.prototype.toPadLeft = function(str) {
            //this 就相当于上面的num
            //str代表要补位的格式
            if (this.toString().length >= str.length) {
                return this.toString();
            } else {
                var count = str.length - this.toString().length; //要补位的长度
                var prefixerStr = ""; //存放要补多少个0
                for (var i = 0; i < count; i++) {
                    prefixerStr += "0";
                }
                return prefixerStr + this.toString();
            }
        }
        //这个参数就是要格式化的值    下面的filter就是我们定义的过滤器
    function toDate(input) {
        //判断一下这个地方的input的类型 ，如果是date对象，我什么都不做 如果是字符串 我就要转成date对象
        var d = null;
        if (input instanceof Date) {
            //说明 它是通过Date 来new 出来的实例
            d = input;
        } else {
            //就有可能是string类型
            d = new Date(input);
        }
        var year = d.getFullYear();
        var month = d.getMonth().toPadLeft("00");
        var date = d.getDate().toPadLeft("00");

        var hour = d.getHours().toPadLeft("00");
        var minute = d.getMinutes().toPadLeft("00");
        var second = d.getSeconds().toPadLeft("00");

        //要求：2018-03-10 15:43:06
        //return `${year}-${month}-${date} ${hour}:${minute}:${second}`;
        return [year, month, date].join("-") + " " + [hour, minute, second].join(":");
    }
    var app = new Vue({
        //在Vue1.0的版本里面，它可以是html,也可以是body标签，但是在2.0里面，取消了
        //EL它是一个选择器，但是一定要注意，不能是body
        el: ".app",
        data: {
            dateNow: new Date(),
            endDate: "Tue Apr 10 2018 15:53:11 GMT+0800 (中国标准时间)"
        },
        computed: {

        },
        filters: {
            //局步的过滤器
            toDate
        }
    })
</script>
