  4. 如果返回一个前端的视图

    根据我们MVC的概念，我们用户的请求是最先到达我们的一个控制器里面的方法app.get() router.get()这个里面的，根据这一个请求的方法，我们可以在这里做一些相应的操作，例如，重定向resp.redirect(""),或者是返回一个json对象到前端resp.json(),现在如果要返回一个视图，那么，我们就需要去编写视图文件，这个时候，我们会在views这个文件夹的下面去创建我们的视图文件，然后调用我们的resp.render()这一个方法去返回一个视图到前面

    app.get("/login", (req, resp) => {
    resp.render("login");
});

    resp.render("login",{stuList:["张三","李四"]});

# Express里面路由的设置

首先，我们要弄清楚一点，我们为什么需要去使用路由

在编写app.js的代码的时候，我们发现，如果把所有的后台请求全部都放在app.js这个文件里面，我们就发现，这个文件会变得越来越大，不容易管理，如果我们想解决这个问题，把代码就需要进行拆分，拆分这个代码的技术，我们就叫做路由

  1. 首先先编写路由文件，把之前的请求分模块编写路由文件

    const express = require('express');
const router = express.Router();
router.get("/login", (req, resp) => {
    resp.render("User/login");
    //代表的就是返回User目录下面的login视图
});
//如果请求地址同名了，只要请求方式不一样，都没问题
router.post("/login", (req, resp) => {
    //如果是post的请求，我们可以在这个地方通过req.body这个属性拿到值 
    //后台接收的值是根据表单里面的name这个属性来取值的
    // var userName = req.body.userName;
    // var pwd = req.body.pwd;
    var params = { userName, pwd } = req.body;
    console.log(userName, pwd);
    resp.json(params);
});
module.exports = router;

现在，我们就针对数据库操作来进行一个封装，这个封装过程是通用的，在其它的程序里面（C#,Java,PHP）都适用

  1. 编写数据库连接的对象，用于获取数据库的连接，自己去看DBUtil的封装
  2. 根据不同的数据库表来进行不同的模块封装，因为基本上所有的数据库都有四个基本操作【增删改查】

function sendMail(to, content) {
    return new Promise((resolve, reject) => {

            service: "qq",
            auth: {
                user: "365055754@qq.com",

            to: to,
            from: "365055754@qq.com",
            subject: "注册验证码",
            text: content,
            html: content
        }, (err, info) => {
            if (err) {
                reject(err); //catch
            } else {
                resolve(info); //then
            }
        });
    })
}
module.exports = sendMail;

      const DBUtil = require('../utils/DBUtil');
class VCodeService {
    addVCode({ email, vcode, vtime }) {
        return new Promise((resolve, reject) => {
            var conn = DBUtil.getConn();
            conn.query("insert into vcode (vcode,email,vtime) values (?,?,?)", [vcode, email, vtime], (err, result) => {
                if (err) {
                    reject(err);
                } else {
                    if (result.affectedRows > 0) {
                        resolve(true);
                    } else {
                        resolve(false);
                    }
                }
            });
            conn.end();
        });
    }
}
module.exports = VCodeService;
